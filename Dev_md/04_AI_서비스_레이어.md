## Provider 추상화 - AsyncGenerator 패턴

모든 AI Provider는 AsyncGenerator 패턴을 사용하여 스트리밍 응답을 처리한다.

```javascript
// 각 Provider는 AsyncGenerator를 반환
async function* streamResponse(apiKey, messages, model, options) {
  const response = await fetch(endpoint, { ... });
  const reader = response.body.getReader();
  const decoder = new TextDecoder();

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;

    const chunk = decoder.decode(value);
    // SSE 파싱 후 텍스트 청크 yield
    yield parsedText;
  }
}
```

## Factory 패턴 (provider-factory.js)

```javascript
// provider-factory.js
export function createProvider(providerName) {
  switch (providerName) {
    case 'openai':
      return openaiProvider;
    case 'claude':
      return claudeProvider;
    case 'gemini':
      return geminiProvider;
    case 'genspark':
      return gensparkProvider;
    default:
      throw new Error(`Unknown provider: ${providerName}`);
  }
}
```

## Provider별 스트리밍 SSE 구현

### OpenAI Provider

```javascript
// openai-provider.js
const ENDPOINT = 'https://api.openai.com/v1/chat/completions';

async function* stream(apiKey, messages, model) {
  const response = await fetch(ENDPOINT, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${apiKey}`,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      model,           // 'gpt-4o' or 'gpt-4o-mini'
      messages,
      stream: true,
    }),
  });

  // SSE 형식: data: {"choices":[{"delta":{"content":"텍스트"}}]}
  for await (const chunk of parseSSE(response)) {
    const content = chunk.choices?.[0]?.delta?.content;
    if (content) yield content;
  }
}
```

```javascript
// claude-provider.js
const ENDPOINT = 'https://api.anthropic.com/v1/messages';

async function* stream(apiKey, messages, model) {
  const response = await fetch(ENDPOINT, {
    method: 'POST',
    headers: {
      'x-api-key': apiKey,
      'Content-Type': 'application/json',
      'anthropic-version': '2023-06-01',
      'anthropic-dangerous-direct-browser-access': 'true',
    },
    body: JSON.stringify({
      model,           // 'claude-sonnet-4-20250514' or 'claude-haiku-4-20250414'
      messages,
      stream: true,
      max_tokens: 4096,
    }),
  });

  // SSE 형식: event: content_block_delta, data: {"delta":{"text":"텍스트"}}
  for await (const chunk of parseSSE(response)) {
    if (chunk.type === 'content_block_delta') {
      yield chunk.delta.text;
    }
  }
}
```

```javascript
    if (text) yield text;

}
```

## 토큰 추정

```javascript
// tokenEstimator.js
export function estimateTokens(text) {
  // 영어: ~4 characters per token
  // 한국어: ~2 characters per token
  const englishChars = text.replace(/[^\x00-\x7F]/g, '').length;
  const koreanChars = text.length - englishChars;

  const englishTokens = Math.ceil(englishChars / 4);
  const koreanTokens = Math.ceil(koreanChars / 2);

  return englishTokens + koreanTokens;
}
```

## 비용 계산

```javascript
// costCalculator.js
const PRICING = {
  'gpt-4o': { input: 2.50, output: 10.00 },        // per 1M tokens
  'gpt-4o-mini': { input: 0.15, output: 0.60 },
  'claude-sonnet-4': { input: 3.00, output: 15.00 },
  'claude-haiku-4.5': { input: 0.80, output: 4.00 },
  'gemini-2.0-flash': { input: 0.10, output: 0.40 },
  'gemini-2.5-pro': { input: 1.25, output: 10.00 },
};

export function calculateCost(model, inputTokens, outputTokens) {
  const pricing = PRICING[model];
  const inputCost = (inputTokens / 1_000_000) * pricing.input;
  const outputCost = (outputTokens / 1_000_000) * pricing.output;
  return inputCost + outputCost;
}
```

## 재시도 로직 (Exponential Backoff)

```javascript
async function* streamWithRetry(provider, apiKey, messages, model, maxRetries = 3) {
  let attempt = 0;

  while (attempt < maxRetries) {
    try {
      yield* provider.stream(apiKey, messages, model);
      return; // 성공 시 종료
    } catch (error) {
      attempt++;
      if (attempt >= maxRetries) throw error;

      // Exponential backoff: 1s, 2s, 4s
      const delay = Math.pow(2, attempt - 1) * 1000;
      await new Promise(resolve => setTimeout(resolve, delay));
    }
  }
}
```

## useAIStream Hook

```javascript
// useAIStream.js
export function useAIStream() {
  const [output, setOutput] = useState('');
  const [isStreaming, setIsStreaming] = useState(false);
  const [error, setError] = useState(null);
  const [tokenCount, setTokenCount] = useState({ input: 0, output: 0 });
  const abortControllerRef = useRef(null);

  const startStream = useCallback(async (provider, apiKey, messages, model) => {
    setIsStreaming(true);
    setOutput('');
    setError(null);
    abortControllerRef.current = new AbortController();

    try {
      const providerInstance = createProvider(provider);
      let fullText = '';

      for await (const chunk of providerInstance.stream(apiKey, messages, model)) {
        fullText += chunk;
        setOutput(fullText);
      }

      setTokenCount({
        input: estimateTokens(JSON.stringify(messages)),
        output: estimateTokens(fullText),
      });
    } catch (err) {
      if (err.name !== 'AbortError') {
        setError(err.message);
      }
    } finally {
      setIsStreaming(false);
    }
  }, []);

  const stopStream = useCallback(() => {
    abortControllerRef.current?.abort();
    setIsStreaming(false);
  }, []);

  return { output, isStreaming, error, tokenCount, startStream, stopStream };
}
```

## useApiKeys Hook

```javascript
// useApiKeys.js
export function useApiKeys() {
  const [apiKeys, setApiKeys] = useState({});

  // localStorage에서 로드 + Supabase와 동기화
  useEffect(() => {
    const localKeys = JSON.parse(localStorage.getItem('api_keys') || '{}');
    setApiKeys(localKeys);

    // Supabase에서 동기화
    syncWithSupabase(localKeys);
  }, []);

  const saveApiKey = (provider, key) => {
    const updated = { ...apiKeys, [provider]: key };
    setApiKeys(updated);
    localStorage.setItem('api_keys', JSON.stringify(updated));

    // Supabase에도 저장
    saveToSupabase(updated);
  };

  const getApiKey = (provider) => apiKeys[provider] || '';

  return { apiKeys, saveApiKey, getApiKey };
}
```
